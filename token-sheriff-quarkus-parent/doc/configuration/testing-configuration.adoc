== Purpose

Testing configuration patterns and utilities for applications using the Token-Sheriff Quarkus extension, covering test profiles, token generation, and validation testing.

== Test Profile Configuration

=== Basic Test Profile Setup

[source,java]
----
public class JwtTestProfile implements QuarkusTestProfile {
    @Override
    public Map<String, String> getConfigOverrides() {
        Map<String, String> config = new HashMap<>();
        config.put("sheriff.token.issuers.default.issuer-identifier", "https://test.example.com/auth");
        config.put("sheriff.token.issuers.default.enabled", "true");
        config.put("sheriff.token.issuers.default.jwks.file-path", "classpath:test-jwks.json");
        config.put("sheriff.token.issuers.default.expected-audience", "test-application");
        return config;
    }
}
----

@QuarkusTest

public class MultiIssuerTestProfile implements QuarkusTestProfile {
    @Override

        config.put("sheriff.token.issuers.primary.expected-audience", "primary-app");
        config.put("sheriff.token.issuers.secondary.jwks.http.well-known-url", "https://secondary-auth.example.com/.well-known/openid-configuration");
        config.put("sheriff.token.issuers.secondary.enabled", "true");
        config.put("sheriff.token.issuers.secondary.expected-audience", "secondary-app");
        config.put("sheriff.token.parser.max-token-size", "16384");
        return config;
    }
}

== Test Token Generation

The `TestTokenHolder` class generates signed JWT tokens for testing. It takes a `TokenType` and a `ClaimControlParameter` in its constructor, generates claims using built-in generators, and signs the token with an in-memory RSA key pair.

=== Basic Token Creation

[source,java]
----
@Test
void shouldValidateBasicToken() {
    // Create a token holder with default (valid) claims for ACCESS_TOKEN
    var claimControl = ClaimControlParameter.builder().build();
    var tokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);

    // Get a matching IssuerConfig with the correct public key material
    IssuerConfig issuerConfig = tokenHolder.getIssuerConfig();

    // Build a TokenValidator using the issuer config
    TokenValidator tokenValidator = TokenValidator.builder()
        .issuerConfig(issuerConfig)
        .build();

    // Validate the token — returns AccessTokenContent on success
    AccessTokenContent result = tokenValidator.createAccessToken(
        AccessTokenRequest.of(tokenHolder.getRawToken()));
    assertNotNull(result);
}
----

=== Expired Token Testing

[source,java]
----
@Test
void shouldRejectExpiredToken() {
    // Create a token holder with an expired token
    var claimControl = ClaimControlParameter.builder()
        .expiredToken(true)
        .build();
    var tokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);

    IssuerConfig issuerConfig = tokenHolder.getIssuerConfig();
    TokenValidator tokenValidator = TokenValidator.builder()
        .issuerConfig(issuerConfig)
        .build();

    // Validation throws TokenValidationException for expired tokens
    assertThrows(TokenValidationException.class, () ->
        tokenValidator.createAccessToken(
            AccessTokenRequest.of(tokenHolder.getRawToken())));
}
----

=== Invalid Signature Testing

[source,java]
----
@Test
void shouldRejectInvalidSignature() {
    // Create a valid token
    var claimControl = ClaimControlParameter.builder().build();
    var tokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);

    // Use a DIFFERENT key material for the issuer config (signature mismatch)
    var differentTokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);
    IssuerConfig issuerConfig = differentTokenHolder.getIssuerConfig();

    TokenValidator tokenValidator = TokenValidator.builder()
        .issuerConfig(issuerConfig)
        .build();

    // Validation throws TokenValidationException for signature mismatch
    assertThrows(TokenValidationException.class, () ->
        tokenValidator.createAccessToken(
            AccessTokenRequest.of(tokenHolder.getRawToken())));
}

== Testing Patterns

=== Token Claim Manipulation

[source,java]
----
@Test
void shouldValidateTokenWithCustomClaims() {
    var claimControl = ClaimControlParameter.builder().build();
    var tokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);

    // Mutate claims using the fluent API
    tokenHolder.withClaim("custom_claim", ClaimValue.forPlainString("custom_value"));

    IssuerConfig issuerConfig = tokenHolder.getIssuerConfig();
    TokenValidator tokenValidator = TokenValidator.builder()
        .issuerConfig(issuerConfig)
        .build();

    AccessTokenContent result = tokenValidator.createAccessToken(
        AccessTokenRequest.of(tokenHolder.getRawToken()));
    assertNotNull(result);
}
----

@Test
void shouldRejectUnknownIssuer() {
    // Create a token with a specific issuer
    var claimControl = ClaimControlParameter.builder().build();
    var tokenHolder = new TestTokenHolder(TokenType.ACCESS_TOKEN, claimControl);

    // Override the issuer claim to one not matching the IssuerConfig
    tokenHolder.withClaim("iss", ClaimValue.forPlainString("https://unknown-issuer.example.com"));

    IssuerConfig issuerConfig = tokenHolder.getIssuerConfig();
    TokenValidator tokenValidator = TokenValidator.builder()
        .issuerConfig(issuerConfig)
        .build();

    assertThrows(TokenValidationException.class, () ->
        tokenValidator.createAccessToken(
            AccessTokenRequest.of(tokenHolder.getRawToken())));
